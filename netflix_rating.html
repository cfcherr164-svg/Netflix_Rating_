# Method 1: Using nbconvert (most common for Jupyter Notebooks)
import nbconvert
from nbconvert import HTMLExporter
import os

# Convert Jupyter Notebook to HTML
def convert_notebook_to_html(notebook_path, output_path=None):
    """Convert a Jupyter Notebook to HTML file"""
    
    # Create HTML exporter
    html_exporter = HTMLExporter()
    
    # Read and convert the notebook
    with open(notebook_path, 'r', encoding='utf-8') as f:
        notebook_content = f.read()
    
    # Convert to HTML
    (body, resources) = html_exporter.from_filename(notebook_path)
    
    # Set output path
    if output_path is None:
        output_path = notebook_path.replace('.ipynb', '.html')
    
    # Write HTML file
    with open(output_path, 'w', encoding='utf-8') as f:
        f.write(body)
    
    print(f"Notebook converted to: {output_path}")
    return output_path

# Method 3: For Python code to HTML with syntax highlighting
from pygments import highlight
from pygments.lexers import PythonLexer
from pygments.formatters import HtmlFormatter

def python_code_to_html(code_string, output_file):
    """Convert Python code string to HTML with syntax highlighting"""
    
    # Create HTML with syntax highlighting
    formatter = HtmlFormatter(style='default', full=True)
    highlighted_code = highlight(code_string, PythonLexer(), formatter)
    
    # Write to file
    with open(output_file, 'w', encoding='utf-8') as f:
        f.write(highlighted_code)
    
    print(f"Code converted to HTML: {output_file}")
